document.addEventListener("DOMContentLoaded", () => { // Sélection du tableau const tbody = document.getElementById("biens-tbody"); if (!tbody) { console.warn("⚠️ biens-tbody introuvable dans proprietes.html"); return; } // Exemple : récupération des biens (à adapter selon ton système) import("/js/core/modules/entreprises.js").then(module => { const entreprises = module.entreprises || {}; tbody.innerHTML = ""; Object.values(entreprises).forEach(bien => { const tr = document.createElement("tr"); tr.innerHTML = ` ${bien.nom || "?"} ${bien.categorie || "?"} ${bien.prixAchat?.toLocaleString("fr-FR") || 0} € ${bien.loyer?.toLocaleString("fr-FR") || 0} € ${bien.chargesSociales?.toLocaleString("fr-FR") || 0} € ${bien.impots?.toLocaleString("fr-FR") || 0} € ${(bien.loyer - bien.chargesSociales - bien.impots).toLocaleString("fr-FR")} € `; tbody.appendChild(tr); }); }).catch(err => { console.error("Erreur lors du chargement des biens :", err); }); });
